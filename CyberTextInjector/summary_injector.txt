3. JS (Logic)
function processNextChar() {
    // Handling glitched decoding effect
    const glitchInterval = setInterval(() => {
        cycles++;
        if (cycles < maxCycles) {
            // Show random symbol
            currentTextObj[currentIndex].char = symbols[Math.floor(Math.random() * symbols.length)];
            if(Math.random() > 0.5) playClick(); // Sound effect
        } else {
            // Reveal actual char
            currentTextObj[currentIndex].char = targetChar;
            currentTextObj[currentIndex].isDecoded = true;
            clearInterval(glitchInterval);
            currentIndex++;
            processNextChar(); // Recursion for next char
        }
        renderHTML(currentTextObj);
    }, 15);
}
